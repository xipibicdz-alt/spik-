var keepAliveTimer=null;
function speak(txt){
  if(!txt||!txt.trim())return;
  if(keepAliveTimer){clearInterval(keepAliveTimer);keepAliveTimer=null;}
  window.speechSynthesis.cancel();
  setTimeout(function(){
    var u=new SpeechSynthesisUtterance(txt.trim());
    u.lang='ar-SA';
    u.rate=parseFloat(vp.rate);
    u.pitch=parseFloat(vp.pitch);
    u.volume=parseFloat(vp.vol);
    var v=getVoice();if(v)u.voice=v;
    u.onstart=function(){sbars.classList.add('show');btnSpeak.classList.add('pulse');};
    u.onend=function(){sbars.classList.remove('show');btnSpeak.classList.remove('pulse');if(keepAliveTimer){clearInterval(keepAliveTimer);keepAliveTimer=null;}};
    u.onerror=function(){sbars.classList.remove('show');btnSpeak.classList.remove('pulse');if(keepAliveTimer){clearInterval(keepAliveTimer);keepAliveTimer=null;}};
    window.speechSynthesis.speak(u);
    // Chrome keepalive fix
    keepAliveTimer=setInterval(function(){
      if(!window.speechSynthesis.speaking){clearInterval(keepAliveTimer);keepAliveTimer=null;return;}
      window.speechSynthesis.pause();
      window.speechSynthesis.resume();
    },9000);
  },50);
}
function stopSpeak(){
  if(keepAliveTimer){clearInterval(keepAliveTimer);keepAliveTimer=null;}
  window.speechSynthesis.cancel();
  sbars.classList.remove('show');
  btnSpeak.classList.remove('pulse');
}

/* ── SHORTCUTS ── */
function applyFull(text){
  if(!scOn)return text;
  for(var i=0;i<shortcuts.length;i++){if(text.trim()===shortcuts[i].k)return shortcuts[i].v;}
  return text;
}
function checkInline(){
  if(!scOn)return;
  var text=ta.value, pos=ta.selectionStart;
  var before=text.slice(0,pos);
  var parts=before.split(/[\s،.]+/);
  var word='';
  for(var i=parts.length-1;i>=0;i--){var w=parts[i].trim();if(w){word=w;break;}}
  if(!word)return;
  var sc=null;
  for(var j=0;j<shortcuts.length;j++){if(shortcuts[j].k===word){sc=shortcuts[j];break;}}
  if(!sc)return;
  var idx=before.lastIndexOf(word);if(idx<0)return;
  var newText=text.slice(0,idx)+sc.v+text.slice(idx+word.length);
  ta.value=newText;
  var np=idx+sc.v.length;ta.setSelectionRange(np,np);
  updateCC();
  if(ttsOn)speak(sc.v);
  toast('⚡ اختصار: '+sc.k,'ok');
}
